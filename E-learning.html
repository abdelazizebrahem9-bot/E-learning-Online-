// ============================================================ // Google Apps Script — حفظ بيانات التسجيل في Google Sheets // انسخ الكود ده كله في Apps Script وانشره // ============================================================ function doPost(e) { try { // افتح الـ Spreadsheet الحالي const sheet = SpreadsheetApp.getActiveSpreadsheet().getActiveSheet(); // أضف عناوين الأعمدة إذا الشيت فاضي if (sheet.getLastRow() === 0) { sheet.appendRow([ "التاريخ", "الاسم الكامل", "السن", "البريد الإلكتروني", "الجامعة", "العنوان", "الكورس" ]); // تنسيق السطر الأول (Header) const headerRange = sheet.getRange(1, 1, 1, 7); headerRange.setBackground("#00c896"); headerRange.setFontColor("#ffffff"); headerRange.setFontWeight("bold"); headerRange.setHorizontalAlignment("center"); } // استخراج البيانات من الفورم const name = e.parameter.name || ""; const age = e.parameter.age || ""; const email = e.parameter.email || ""; const university = e.parameter.university || ""; const address = e.parameter.address || ""; const course = e.parameter.course || ""; const date = e.parameter.date || new Date().toLocaleString(); // أضف صف جديد sheet.appendRow([date, name, age, email, university, address, course]); // رد بنجاح return ContentService .createTextOutput(JSON.stringify({ status: "success" })) .setMimeType(ContentService.MimeType.JSON); } catch (err) { return ContentService .createTextOutput(JSON.stringify({ status: "error", message: err.message })) .setMimeType(ContentService.MimeType.JSON); } } // دالة اختبار (اختياري) function doGet(e) { return ContentService .createTextOutput("✅ Apps Script يعمل بشكل صحيح!") .setMimeType(ContentService.MimeType.TEXT); }
